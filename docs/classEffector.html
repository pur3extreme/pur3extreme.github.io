Effector Class


public class ContainerEffector
{
	int x {get;set;}
	int y {get;set;}
	Point position {get;set;}
	int turnsAlive {get;set;}
	int maxTurnsAlive {get;set;}
	bool active {get;set;}
	EffectorType type => get;

	//Functioins
	void destroy()
	void turnPassed() //May destroy its self
}
