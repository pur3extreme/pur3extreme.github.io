Summon Class


class Summon
{
	object spell; //Default  = null; //if null it spawns a player creature
	Point position; //Default = new Point(750,750);
	int team; //Default = 1;
	bool onPlayersPanel; //Default  = false;
	bool useAI; //Default = true; - ONLY valid if onPlayersPanel == true
        bool no_AI_still_use_turn; //Default = false; ONLY valid if useAI == true //Only used if you have some sort of AI yourself
	string name; //Default  = "";
	bool playSound; //Default  = true;
	Table colors; //Default  = null;
	Table spells; //Default  = null;
	Table outfit; //Default  = null;
	int elemental; //Default  = -1;

	//Alias = "summon()"
	static Summon construct(object spell = null, Point point = null, int team = 1, Table colors = null, bool playSound = true, bool onPlayersPanel = false, string name = "", Table outfit = null, Table spells = null, int elemental = -1)
}

Example

function main()
	toSummon = summon()
	toSummon.onPlayersPanel = true
	toSummon.name = "Jeff"
	toSummon.spells = {"Fire Ball"}
	toSummon.outfit = {73, 0, 24, 0, 49, 55, 0, 0, 0}
	toSummon.colors = {"#FFCC00", "#FF00FF", "#FFFF33", "#FFCC99"}
	game.createCreature(toSummon)
	toSummon.position = point(900, 500)
	toSummon.team = 2
	toSummon.name = "Vern"
	toSummon.outfit = {25, 11, 14, 24, 7, 5, 0, 0, 0}
	toSummon.colors = {"#5300FF", "#405B68", "#DE48EE", "#7C2346"}
	toSummon.spells = {"Arcane Arrows"}
	game.createCreature(toSummon)
	
	--Next creature
	toSummon = summon(SpellEnum.Summon_Dwarf, point(350, 850), game.getPlayerCount()-1)
	sum = game.createCreature(toSummon)
	
	game.ShowInfo(sum.name);
end
